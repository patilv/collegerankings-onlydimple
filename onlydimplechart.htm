1. library(rCharts)
  3. CompRank=read.csv("OverallRankings.csv", stringsAsFactors=FALSE)
  4. BestValue=read.csv("2013BestValues.csv",stringsAsFactors=FALSE)
  5. # Merge both
  6. TotalRank=merge(CompRank,BestValue,all=T)
  7. TotalRank$Name=paste(TotalRank$Name,TotalRank$State,sep=", ")
  8. TotalRank=TotalRank[-c(2)]
  10. # Replacing factor value and converting to numeric
  12. for (i in 1:18){TotalRank[,i]<-gsub("600-650", "625", TotalRank[,i])}
  13. for (i in 1:18){TotalRank[,i]<-gsub("#NA", "", TotalRank[,i])}
  15. for (i in 4:18){TotalRank[,i] = as.numeric(TotalRank[,i])}
  16. for (i in 1:3){TotalRank[,i] = factor(TotalRank[,i])}
  18. TotalRank$Carnegie.Classification=factor(TotalRank$Carnegie.Classification,
  19.                                          c("Baccalaureate Colleges--Arts & Sciences",
  20.                                            "Baccalaureate Colleges--Diverse Fields",
  21.                                            "Master's Colleges and Universities (smaller programs)",
  22.                                            "Master's Colleges and Universities (medium programs)",
  23.                                            "Master's Colleges and Universities (larger programs)",
  24.                                            "Doctoral/Research Universities",
  25.                                            "Research Universities (high research activity)",
  26.                                            "Research Universities (very high research activity)"))
  28. # Overall Rankings of Colleges for those ranked as Best Value
  30. BvalandOrankmore=TotalRank[c(1,2,3,17,18)]
  31. BvalandOrankmore=subset(BvalandOrankmore,BvalandOrankmore$Best.Value.Rank!="NA")
  33. BvalandOrankmore$Name=reorder(BvalandOrankmore$Name,BvalandOrankmore$Best.Value.Rank)
  35. d1 <- dPlot(
  36.   Best.Value.Rank ~ Overall.Rank,
  37.   groups = c("Name","Carnegie.Classification","Control"),
  38.   data = BvalandOrankmore,
  39.   type = "bubble",height=800,
  40.   width=1000,
  41.   bounds = list(x=60, y=25, width=400, height=350)
  42. )
  43. d1$xAxis( type = "addMeasureAxis" )
  44. d1$yAxis( type = "addMeasureAxis" )
  45. d1$legend(
  46.   x = 465,
  47.   y = 0,
  48.   width = 50,
  49.   height = 200,
  50.   horizontalAlign = "left"
  51. )
  52. d1
